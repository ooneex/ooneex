import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import { Assert } from "@ooneex/validation";
import { DeleteFolderService } from "../services/DeleteFolderService";

type DeleteFolderRouteType = {
  params: { id: string };
  response: { id: string };
};

@Route.Delete("/folders/:id", {
  name: "folder.delete",
  version: 1,
  description: "Delete a folder",
  params: {
    id: Assert("string"),
  },
  response: Assert({
    id: "string",
  }),
  roles: [ERole.USER],
})
export class DeleteFolderController {
  constructor(
    @inject(DeleteFolderService) private readonly service: DeleteFolderService,
  ) {}

  public async index(context: ContextType<DeleteFolderRouteType>) {
    const { id } = context.params;

    await this.service.execute({ id });

    return context.response.json({ id });
  }
}
