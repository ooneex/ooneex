import { inject } from "@ooneex/container";
import { decorator } from "@ooneex/service";
import type { IService } from "@ooneex/service";
import type { LocaleType } from "@ooneex/translation";
import { TagEntity } from "../entities/TagEntity";
import { TagRepository } from "../repositories/TagRepository";

type ServiceDataType = {
  name: string;
  lang?: LocaleType;
  color?: string;
  description?: string;
};

@decorator.service()
export class CreateTagService implements IService {
  constructor(
    @inject(TagRepository) private readonly repository: TagRepository,
  ) {}

  public async execute(data?: ServiceDataType): Promise<TagEntity | null> {
    if (!data) return null;

    const tag = new TagEntity();
    tag.name = data.name;
    tag.lang = data.lang;
    tag.color = data.color;
    tag.description = data.description;

    return await this.repository.create(tag);
  }
}
