import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import { Assert } from "@ooneex/validation";
import { DeleteTagService } from "../services/DeleteTagService";

type DeleteTagRouteType = {
  params: { id: string };
  response: { id: string };
};

@Route.Delete("/tags/:id", {
  name: "tag.delete",
  version: 1,
  description: "Delete a tag",
  params: {
    id: Assert("string"),
  },
  response: Assert({
    id: "string",
  }),
  roles: [ERole.USER],
})
export class DeleteTagController {
  constructor(
    @inject(DeleteTagService) private readonly service: DeleteTagService,
  ) {}

  public async index(context: ContextType<DeleteTagRouteType>) {
    const { id } = context.params;

    await this.service.execute({ id });

    return context.response.json({ id });
  }
}
