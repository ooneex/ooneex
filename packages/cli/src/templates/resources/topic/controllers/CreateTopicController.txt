import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import type { LocaleType } from "@ooneex/translation";
import { Assert } from "@ooneex/validation";
import { CreateTopicService } from "../services/CreateTopicService";
import type { TopicEntity } from "../entities/TopicEntity";

type CreateTopicRouteType = {
  payload: { name: string; lang?: LocaleType; color?: string; description?: string };
  response: TopicEntity | null;
};

@Route.post("/topics", {
  name: "topic.create",
  version: 1,
  description: "Create a new topic",
  payload: Assert({
    name: "string",
    lang: "string?",
    color: "string?",
    description: "string?",
  }),
  response: Assert({
    id: "string",
    name: "string",
    lang: "string",
    color: "string?",
    description: "string?",
  }),
  roles: [ERole.USER],
})
export class CreateTopicController {
  constructor(
    @inject(CreateTopicService) private readonly service: CreateTopicService,
  ) {}

  public async index(context: ContextType<CreateTopicRouteType>) {
    const { name, lang, color, description } = context.payload;

    const topic = await this.service.execute({ name, lang, color, description });

    return context.response.json(topic);
  }
}
