import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import type { LocaleType } from "@ooneex/translation";
import { Assert } from "@ooneex/validation";
import { UpdateVideoService } from "../services/UpdateVideoService";
import type { VideoEntity } from "../entities/VideoEntity";

type UpdateVideoRouteType = {
  params: { id: string };
  payload: {
    title?: string;
    src?: string;
    thumbnail?: string;
    color?: string;
    description?: string;
    tags?: string[];
    topics?: string[];
    categories?: string[];
    status?: string;
    lang?: LocaleType;
  };
  response: VideoEntity | null;
};

@Route.patch("/videos/:id", {
  name: "video.update",
  version: 1,
  description: "Update a video",
  params: {
    id: Assert("string"),
  },
  payload: Assert({
    title: "string?",
    src: "string?",
    thumbnail: "string?",
    color: "string?",
    description: "string?",
    tags: "string[]?",
    topics: "string[]?",
    categories: "string[]?",
    status: "string?",
    lang: "string?",
  }),
  response: Assert({
    id: "string",
    title: "string",
    src: "string",
    thumbnail: "string",
    color: "string?",
    description: "string?",
    tags: "string[]?",
    topics: "string[]?",
    categories: "string[]?",
    status: "string?",
    lang: "string?",
  }),
  roles: [ERole.USER],
})
export class UpdateVideoController {
  constructor(
    @inject(UpdateVideoService) private readonly service: UpdateVideoService,
  ) {}

  public async index(context: ContextType<UpdateVideoRouteType>) {
    const { id } = context.params;
    const { title, src, thumbnail, color, description, tags, topics, categories, status, lang } = context.payload;

    const video = await this.service.execute({ id, title, src, thumbnail, color, description, tags, topics, categories, status, lang });

    return context.response.json(video);
  }
}
