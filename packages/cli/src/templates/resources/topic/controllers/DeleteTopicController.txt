import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import { Assert } from "@ooneex/validation";
import { DeleteTopicService } from "../services/DeleteTopicService";

type DeleteTopicRouteType = {
  params: { id: string };
  response: { id: string };
};

@Route.Delete("/topics/:id", {
  name: "topic.delete",
  version: 1,
  description: "Delete a topic",
  params: {
    id: Assert("string"),
  },
  response: Assert({
    id: "string",
  }),
  roles: [ERole.USER],
})
export class DeleteTopicController {
  constructor(
    @inject(DeleteTopicService) private readonly service: DeleteTopicService,
  ) {}

  public async index(context: ContextType<DeleteTopicRouteType>) {
    const { id } = context.params;

    await this.service.execute({ id });

    return context.response.json({ id });
  }
}
