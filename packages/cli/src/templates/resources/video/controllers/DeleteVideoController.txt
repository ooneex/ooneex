import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import { Assert } from "@ooneex/validation";
import { DeleteVideoService } from "../services/DeleteVideoService";

type DeleteVideoRouteType = {
  params: { id: string };
  response: { id: string };
};

@Route.delete("/videos/:id", {
  name: "video.delete",
  version: 1,
  description: "Delete a video",
  params: {
    id: Assert("string"),
  },
  response: Assert({
    id: "string",
  }),
  roles: [ERole.USER],
})
export class DeleteVideoController {
  constructor(
    @inject(DeleteVideoService) private readonly service: DeleteVideoService,
  ) {}

  public async index(context: ContextType<DeleteVideoRouteType>) {
    const { id } = context.params;

    await this.service.execute({ id });

    return context.response.json({ id });
  }
}
