import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import { Assert } from "@ooneex/validation";
import { DeleteCategoryService } from "../services/DeleteCategoryService";

type DeleteCategoryRouteType = {
  params: { id: string };
  response: { id: string };
};

@Route.delete("/categories/:id", {
  name: "category.delete",
  version: 1,
  description: "Delete a category",
  params: {
    id: Assert("string"),
  },
  response: Assert({
    id: "string",
  }),
  roles: [ERole.USER],
})
export class DeleteCategoryController {
  constructor(
    @inject(DeleteCategoryService) private readonly service: DeleteCategoryService,
  ) {}

  public async index(context: ContextType<DeleteCategoryRouteType>) {
    const { id } = context.params;

    await this.service.execute({ id });

    return context.response.json({ id });
  }
}
