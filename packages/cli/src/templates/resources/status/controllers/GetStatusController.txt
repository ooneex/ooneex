import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import { Assert } from "@ooneex/validation";
import { GetStatusService } from "../services/GetStatusService";
import type { StatusEntity } from "../entities/StatusEntity";

type GetStatusRouteType = {
  params: { id: string };
  response: StatusEntity | null;
};

@Route.get("/statuses/:id", {
  name: "status.get",
  version: 1,
  description: "Get a status by id",
  params: {
    id: Assert("string"),
  },
  roles: [ERole.USER],
})
export class GetStatusController {
  constructor(
    @inject(GetStatusService) private readonly service: GetStatusService,
  ) {}

  public async index(context: ContextType<GetStatusRouteType>) {
    const { id } = context.params;

    const status = await this.service.execute({ id });

    return context.response.json(status);
  }
}
