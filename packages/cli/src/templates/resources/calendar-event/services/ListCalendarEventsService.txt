import { inject } from "@ooneex/container";
import { decorator } from "@ooneex/service";
import type { IService } from "@ooneex/service";
import type { FilterResultType } from "@ooneex/types";
import { CalendarEventEntity } from "../entities/CalendarEventEntity";
import { CalendarEventRepository } from "../repositories/CalendarEventRepository";

type ServiceDataType = {
  page?: number;
  limit?: number;
  q?: string;
};

@decorator.service()
export class ListCalendarEventsService implements IService {
  constructor(
    @inject(CalendarEventRepository) private readonly repository: CalendarEventRepository,
  ) {}

  public async execute(data?: ServiceDataType): Promise<FilterResultType<CalendarEventEntity>> {
    return await this.repository.find({
      page: data?.page,
      limit: data?.limit,
      q: data?.q,
    });
  }
}
