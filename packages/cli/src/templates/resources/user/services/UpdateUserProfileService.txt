import { ClerkAuth } from "@ooneex/auth";
import { inject } from "@ooneex/container";
import { decorator } from "@ooneex/service";
import type { IService } from "@ooneex/service";
import { UserRepository } from "../repositories/UserRepository";

type ServiceDataType = {
  userId: string;
  lastName?: string;
  firstName?: string;
};

@decorator.service()
export class UpdateUserProfileService implements IService {
  constructor(
    @inject(ClerkAuth) private readonly auth: ClerkAuth,
    @inject(UserRepository) private readonly repository: UserRepository,
  ) {}

  public async execute(data?: ServiceDataType): Promise<void> {
    if (!data) return;

    const { userId, lastName, firstName } = data;

    await this.auth.updateUser(userId, {
      lastName,
      firstName,
    });
    await this.repository.updateProfile(userId, { lastName, firstName });
  }
}
