import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import { Assert } from "@ooneex/validation";
import { DeleteImageService } from "../services/DeleteImageService";

type DeleteImageRouteType = {
  params: { id: string };
  response: { id: string };
};

@Route.Delete("/images/:id", {
  name: "image.delete",
  version: 1,
  description: "Delete an image",
  params: {
    id: Assert("string"),
  },
  response: Assert({
    id: "string",
  }),
  roles: [ERole.USER],
})
export class DeleteImageController {
  constructor(
    @inject(DeleteImageService) private readonly service: DeleteImageService,
  ) {}

  public async index(context: ContextType<DeleteImageRouteType>) {
    const { id } = context.params;

    await this.service.execute({ id });

    return context.response.json({ id });
  }
}
