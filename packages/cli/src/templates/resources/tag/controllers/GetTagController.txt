import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import { Assert } from "@ooneex/validation";
import { GetTagService } from "../services/GetTagService";
import type { TagEntity } from "../entities/TagEntity";

type GetTagRouteType = {
  params: { id: string };
  response: TagEntity | Record<string, never>;
};

@Route.get("/tags/:id", {
  name: "tag.get",
  version: 1,
  description: "Get a tag by id",
  params: {
    id: Assert("string"),
  },
  roles: [ERole.USER],
})
export class GetTagController {
  constructor(
    @inject(GetTagService) private readonly service: GetTagService,
  ) {}

  public async index(context: ContextType<GetTagRouteType>) {
    const { id } = context.params;

    const tag = await this.service.execute({ id });

    return context.response.json(tag || {});
  }
}
