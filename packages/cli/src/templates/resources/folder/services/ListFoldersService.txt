import { inject } from "@ooneex/container";
import { decorator } from "@ooneex/service";
import type { IService } from "@ooneex/service";
import type { FilterResultType } from "@ooneex/types";
import { FolderEntity } from "../entities/FolderEntity";
import { FolderRepository } from "../repositories/FolderRepository";

type ServiceDataType = {
  page?: number;
  limit?: number;
  q?: string;
};

@decorator.service()
export class ListFoldersService implements IService {
  constructor(
    @inject(FolderRepository) private readonly repository: FolderRepository,
  ) {}

  public async execute(data?: ServiceDataType): Promise<FilterResultType<FolderEntity>> {
    return await this.repository.find({
      page: data?.page,
      limit: data?.limit,
      q: data?.q,
    });
  }
}
