import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import type { LocaleType } from "@ooneex/translation";
import { Assert } from "@ooneex/validation";
import { CreateCalendarEventService } from "../services/CreateCalendarEventService";

type CreateCalendarEventRouteType = {
  payload: {
    title: string;
    startDate: Date;
    endDate: Date;
    type: string;
    icon?: string;
    description?: string;
    color?: string;
    priority?: string;
    organizer?: { id: string; name: string; email?: string };
    tags?: string[];
    lang?: LocaleType;
  };
  response: {
    id: string;
    title: string;
    startDate: Date;
    endDate: Date;
    type: string;
    icon?: string;
    description?: string;
    color?: string;
    priority?: string;
    organizer?: { id: string; name: string; email?: string };
    tags?: string[];
    lang: LocaleType;
  };
};

@Route.post("/calendar-events", {
  name: "calendar-event.create",
  version: 1,
  description: "Create a new calendar event",
  payload: Assert({
    title: "string",
    startDate: "date",
    endDate: "date",
    type: "string",
    icon: "string?",
    description: "string?",
    color: "string?",
    priority: "string?",
    organizer: "object?",
    tags: "array?",
    lang: "string?",
  }),
  response: Assert({
    id: "string",
    title: "string",
    startDate: "date",
    endDate: "date",
    type: "string",
    icon: "string?",
    description: "string?",
    color: "string?",
    priority: "string?",
    organizer: "object?",
    tags: "array?",
    lang: "string",
  }),
  roles: [ERole.USER],
})
export class CreateCalendarEventController {
  constructor(
    @inject(CreateCalendarEventService) private readonly service: CreateCalendarEventService,
  ) {}

  public async index(context: ContextType<CreateCalendarEventRouteType>) {
    const { title, startDate, endDate, type, icon, description, color, priority, organizer, tags, lang } = context.payload;

    const calendarEvent = await this.service.execute({ title, startDate, endDate, type, icon, description, color, priority, organizer, tags, lang });

    return context.response.json(calendarEvent);
  }
}
