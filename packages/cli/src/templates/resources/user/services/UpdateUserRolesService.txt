import { ClerkAuth } from "@ooneex/auth";
import { inject } from "@ooneex/container";
import type { ERole } from "@ooneex/role";
import { decorator } from "@ooneex/service";
import type { IService } from "@ooneex/service";
import { UserRepository } from "../repositories/UserRepository";

type ServiceDataType = {
  userId: string;
  roles: ERole[];
};

@decorator.service()
export class UpdateUserRolesService<T extends ServiceDataType = ServiceDataType> implements IService<T> {
  constructor(
    @inject(ClerkAuth) private readonly auth: ClerkAuth,
    @inject(UserRepository) private readonly repository: UserRepository,
  ) {}

  public async execute(data?: T): Promise<void> {
    if (!data) return;

    const metadata = await this.auth.getUserMetadata(data.userId);

    await this.auth.updateUserMetadata(data.userId, {
      publicMetadata: { ...metadata.publicMetadata, roles: data.roles },
      privateMetadata: { ...metadata.privateMetadata, roles: data.roles },
    });
    await this.repository.updateRoles(data.userId, data.roles);
  }
}
