import { inject } from "@ooneex/container";
import { decorator } from "@ooneex/service";
import type { IService } from "@ooneex/service";
import type { LocaleType } from "@ooneex/translation";
import { StatusEntity } from "../entities/StatusEntity";
import { StatusRepository } from "../repositories/StatusRepository";

type ServiceDataType = {
  id: string;
  name?: string;
  lang?: LocaleType;
  color?: string;
  description?: string;
};

@decorator.service()
export class UpdateStatusService implements IService {
  constructor(
    @inject(StatusRepository) private readonly repository: StatusRepository,
  ) {}

  public async execute(data?: ServiceDataType): Promise<StatusEntity | null> {
    if (!data) return null;

    const { id, ...fields } = data;

    await this.repository.update({ id, ...fields });

    return await this.repository.findOne(id);
  }
}
