import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import type { LocaleType } from "@ooneex/translation";
import { Assert } from "@ooneex/validation";
import { CreateBookService } from "../services/CreateBookService";

type CreateBookRouteType = {
  payload: {
    title: string;
    color?: string;
    description?: string;
    summarize?: string;
    pageCount?: number;
    src: string;
    coverImage?: string;
    tags?: string[];
    topics?: string[];
    categories?: string[];
    lang: LocaleType;
    status?: string;
  };
  response: {
    id: string;
    title: string;
    color?: string;
    description?: string;
    summarize?: string;
    pageCount?: number;
    src: string;
    coverImage?: string;
    tags?: string[];
    topics?: string[];
    categories?: string[];
    lang: LocaleType;
    status?: string;
  };
};

@Route.Post("/books", {
  name: "book.create",
  version: 1,
  description: "Create a new book",
  payload: Assert({
    title: "string",
    color: "string?",
    description: "string?",
    summarize: "string?",
    pageCount: "number?",
    src: "string",
    coverImage: "string?",
    tags: "string[]?",
    topics: "string[]?",
    categories: "string[]?",
    lang: "string",
    status: "string?",
  }),
  response: Assert({
    id: "string",
    title: "string",
    color: "string?",
    description: "string?",
    summarize: "string?",
    pageCount: "number?",
    src: "string",
    coverImage: "string?",
    tags: "string[]?",
    topics: "string[]?",
    categories: "string[]?",
    lang: "string",
    status: "string?",
  }),
  roles: [ERole.USER],
})
export class CreateBookController {
  constructor(
    @inject(CreateBookService) private readonly service: CreateBookService,
  ) {}

  public async index(context: ContextType<CreateBookRouteType>) {
    const { title, color, description, summarize, pageCount, src, coverImage, tags, topics, categories, lang, status } = context.payload;

    const book = await this.service.execute({ title, color, description, summarize, pageCount, src, coverImage, tags, topics, categories, lang, status });

    return context.response.json(book);
  }
}
