import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import type { FilterResultType } from "@ooneex/types";
import { Assert } from "@ooneex/validation";
import type { NoteEntity } from "../entities/NoteEntity";
import { ListNotesService } from "../services/ListNotesService";

type ListNotesRouteType = {
  query: { page?: number; limit?: number; q?: string };
  response: FilterResultType<NoteEntity>;
};

@Route.get("/notes", {
  name: "note.list",
  version: 1,
  description: "List all notes",
  query: Assert({
    page: "number?",
    limit: "number?",
    q: "string?",
  }),
  roles: [ERole.USER],
})
export class ListNotesController {
  constructor(
    @inject(ListNotesService) private readonly service: ListNotesService,
  ) {}

  public async index(context: ContextType<ListNotesRouteType>) {
    const { page, limit, q } = context.query;

    const result = await this.service.execute({ page, limit, q });

    return context.response.json(result);
  }
}
