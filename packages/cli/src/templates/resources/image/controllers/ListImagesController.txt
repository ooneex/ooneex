import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import type { FilterResultType } from "@ooneex/types";
import { Assert } from "@ooneex/validation";
import type { ImageEntity } from "../entities/ImageEntity";
import { ListImagesService } from "../services/ListImagesService";

type ListImagesRouteType = {
  queries: { page?: number; limit?: number; q?: string };
  response: FilterResultType<ImageEntity>;
};

@Route.get("/images", {
  name: "image.list",
  version: 1,
  description: "List all images",
  queries: Assert({
    page: "number?",
    limit: "number?",
    q: "string?",
  }),
  roles: [ERole.USER],
})
export class ListImagesController {
  constructor(
    @inject(ListImagesService) private readonly service: ListImagesService,
  ) {}

  public async index(context: ContextType<ListImagesRouteType>) {
    const { page, limit, q } = context.queries;

    const result = await this.service.execute({ page, limit, q });

    return context.response.json(result);
  }
}
