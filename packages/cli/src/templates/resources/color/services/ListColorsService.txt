import { inject } from "@ooneex/container";
import { decorator } from "@ooneex/service";
import type { IService } from "@ooneex/service";
import type { FilterResultType } from "@ooneex/types";
import { ColorEntity } from "../entities/ColorEntity";
import { ColorRepository } from "../repositories/ColorRepository";

type ServiceDataType = {
  page?: number;
  limit?: number;
  q?: string;
};

@decorator.service()
export class ListColorsService implements IService {
  constructor(
    @inject(ColorRepository) private readonly repository: ColorRepository,
  ) {}

  public async execute(data?: ServiceDataType): Promise<FilterResultType<ColorEntity>> {
    return await this.repository.find({
      page: data?.page,
      limit: data?.limit,
      q: data?.q,
    });
  }
}
