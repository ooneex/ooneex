import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import type { LocaleType } from "@ooneex/translation";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import { Assert } from "@ooneex/validation";
import { ListCategoriesService } from "../services/ListCategoriesService";

type ListCategoriesRouteType = {
  query: { page?: number; limit?: number; q?: string };
  response: {
    resources: { id: string; name: string; lang: LocaleType; color?: string; description?: string }[];
    total: number;
    totalPages: number;
    page: number;
    limit: number;
  };
};

@Route.get("/categories", {
  name: "category.list",
  version: 1,
  description: "List all categories",
  query: Assert({
    page: "number?",
    limit: "number?",
    q: "string?",
  }),
  roles: [ERole.USER],
})
export class ListCategoriesController {
  constructor(
    @inject(ListCategoriesService) private readonly service: ListCategoriesService,
  ) {}

  public async index(context: ContextType<ListCategoriesRouteType>) {
    const { page, limit, q } = context.query;

    const result = await this.service.execute({ page, limit, q });

    return context.response.json(result);
  }
}
