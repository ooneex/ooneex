import { inject } from "@ooneex/container";
import { decorator } from "@ooneex/service";
import type { IService } from "@ooneex/service";
import type { FilterResultType } from "@ooneex/types";
import { TopicEntity } from "../entities/TopicEntity";
import { TopicRepository } from "../repositories/TopicRepository";

type ServiceDataType = {
  page?: number;
  limit?: number;
  q?: string;
};

@decorator.service()
export class ListTopicsService<T extends ServiceDataType = ServiceDataType> implements IService<T> {
  constructor(
    @inject(TopicRepository) private readonly repository: TopicRepository,
  ) {}

  public async execute(data?: T): Promise<FilterResultType<TopicEntity>> {
    return await this.repository.find({
      page: data?.page,
      limit: data?.limit,
      q: data?.q,
    });
  }
}
