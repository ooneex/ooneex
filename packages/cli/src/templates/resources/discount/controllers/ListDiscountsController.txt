import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import type { FilterResultType } from "@ooneex/types";
import { Assert } from "@ooneex/validation";
import type { DiscountEntity } from "../entities/DiscountEntity";
import { ListDiscountsService } from "../services/ListDiscountsService";

type ListDiscountsRouteType = {
  query: { page?: number; limit?: number; q?: string };
  response: FilterResultType<DiscountEntity>;
};

@Route.get("/discounts", {
  name: "discount.list",
  version: 1,
  description: "List all discounts",
  query: Assert({
    page: "number?",
    limit: "number?",
    q: "string?",
  }),
  roles: [ERole.USER],
})
export class ListDiscountsController {
  constructor(
    @inject(ListDiscountsService) private readonly service: ListDiscountsService,
  ) {}

  public async index(context: ContextType<ListDiscountsRouteType>) {
    const { page, limit, q } = context.query;

    const result = await this.service.execute({ page, limit, q });

    return context.response.json(result);
  }
}
