import { inject } from "@ooneex/container";
import { decorator } from "@ooneex/service";
import type { IService } from "@ooneex/service";
import type { LocaleType } from "@ooneex/translation";
import { NoteEntity } from "../entities/NoteEntity";
import { NoteRepository } from "../repositories/NoteRepository";

type ServiceDataType = {
  id: string;
  title?: string;
  color?: string;
  content?: string;
  description?: string;
  src?: string;
  pageCount?: number;
  coverImage?: string;
  tags?: string[];
  topics?: string[];
  categories?: string[];
  lang?: LocaleType;
  status?: string;
};

@decorator.service()
export class UpdateNoteService implements IService {
  constructor(
    @inject(NoteRepository) private readonly repository: NoteRepository,
  ) {}

  public async execute(data?: ServiceDataType): Promise<NoteEntity | null> {
    if (!data) return null;

    const { id, ...fields } = data;

    await this.repository.update({ id, ...fields });

    return await this.repository.findOne(id);
  }
}
