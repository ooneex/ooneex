import { inject } from "@ooneex/container";
import { decorator } from "@ooneex/service";
import type { IService } from "@ooneex/service";
import { TopicRepository } from "../repositories/TopicRepository";

type ServiceDataType = {
  id: string;
};

@decorator.service()
export class DeleteTopicService<T extends ServiceDataType = ServiceDataType> implements IService<T> {
  constructor(
    @inject(TopicRepository) private readonly repository: TopicRepository,
  ) {}

  public async execute(data?: T): Promise<void> {
    if (!data) return;

    await this.repository.delete({ id: data.id });
  }
}
