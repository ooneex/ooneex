import { inject } from "@ooneex/container";
import { decorator } from "@ooneex/service";
import type { IService } from "@ooneex/service";
import type { FilterResultType } from "@ooneex/types";
import { ImageEntity } from "../entities/ImageEntity";
import { ImageRepository } from "../repositories/ImageRepository";

type ServiceDataType = {
  page?: number;
  limit?: number;
  q?: string;
};

@decorator.service()
export class ListImagesService<T extends ServiceDataType = ServiceDataType> implements IService<T> {
  constructor(
    @inject(ImageRepository) private readonly repository: ImageRepository,
  ) {}

  public async execute(data?: T): Promise<FilterResultType<ImageEntity>> {
    return await this.repository.find({
      page: data?.page,
      limit: data?.limit,
      q: data?.q,
    });
  }
}
