import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import { Assert } from "@ooneex/validation";
import { GetTopicService } from "../services/GetTopicService";
import type { TopicEntity } from "../entities/TopicEntity";

type GetTopicRouteType = {
  params: { id: string };
  response: TopicEntity | Record<string, never>;
};

@Route.get("/topics/:id", {
  name: "topic.get",
  version: 1,
  description: "Get a topic by id",
  params: {
    id: Assert("string"),
  },
  roles: [ERole.USER],
})
export class GetTopicController {
  constructor(
    @inject(GetTopicService) private readonly service: GetTopicService,
  ) {}

  public async index(context: ContextType<GetTopicRouteType>) {
    const { id } = context.params;

    const topic = await this.service.execute({ id });

    return context.response.json(topic || {});
  }
}
