import { inject } from "@ooneex/container";
import { decorator } from "@ooneex/service";
import type { IService } from "@ooneex/service";
import { DiscountEntity } from "../entities/DiscountEntity";
import { DiscountRepository } from "../repositories/DiscountRepository";

type ServiceDataType = {
  label?: string;
  code: string;
  type: "credit" | "access";
  creditAmount?: number;
  accessDuration?: number;
  accessUnit?: "hour" | "day" | "month" | "year";
  maxUsage?: number;
  usageCount?: number;
};

@decorator.service()
export class CreateDiscountService implements IService {
  constructor(
    @inject(DiscountRepository) private readonly repository: DiscountRepository,
  ) {}

  public async execute(data?: ServiceDataType): Promise<DiscountEntity | null> {
    if (!data) return null;

    const discount = new DiscountEntity();
    discount.label = data.label;
    discount.code = data.code;
    discount.type = data.type;
    discount.creditAmount = data.creditAmount;
    discount.accessDuration = data.accessDuration;
    discount.accessUnit = data.accessUnit;
    discount.maxUsage = data.maxUsage;
    discount.usageCount = data.usageCount;

    return await this.repository.create(discount);
  }
}
