import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import { Assert } from "@ooneex/validation";
import { DeleteCalendarEventService } from "../services/DeleteCalendarEventService";

type DeleteCalendarEventRouteType = {
  params: { id: string };
  response: { id: string };
};

@Route.Delete("/calendar-events/:id", {
  name: "calendar-event.delete",
  version: 1,
  description: "Delete a calendar event",
  params: {
    id: Assert("string"),
  },
  response: Assert({
    id: "string",
  }),
  roles: [ERole.USER],
})
export class DeleteCalendarEventController {
  constructor(
    @inject(DeleteCalendarEventService) private readonly service: DeleteCalendarEventService,
  ) {}

  public async index(context: ContextType<DeleteCalendarEventRouteType>) {
    const { id } = context.params;

    await this.service.execute({ id });

    return context.response.json({ id });
  }
}
