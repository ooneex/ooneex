import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import { Assert } from "@ooneex/validation";
import type { FolderEntity } from "../entities/FolderEntity";
import { GetFolderService } from "../services/GetFolderService";

type GetFolderRouteType = {
  params: { id: string };
  response: FolderEntity | Record<string, never>;
};

@Route.get("/folders/:id", {
  name: "folder.get",
  version: 1,
  description: "Get a folder by id",
  params: {
    id: Assert("string"),
  },
  roles: [ERole.USER],
})
export class GetFolderController {
  constructor(
    @inject(GetFolderService) private readonly service: GetFolderService,
  ) {}

  public async index(context: ContextType<GetFolderRouteType>) {
    const { id } = context.params;

    const folder = await this.service.execute({ id });

    return context.response.json(folder || {});
  }
}
