import { inject } from "@ooneex/container";
import { decorator } from "@ooneex/service";
import type { IService } from "@ooneex/service";
import type { FilterResultType } from "@ooneex/types";
import { TagEntity } from "../entities/TagEntity";
import { TagRepository } from "../repositories/TagRepository";

type ServiceDataType = {
  page?: number;
  limit?: number;
  q?: string;
};

@decorator.service()
export class ListTagsService implements IService {
  constructor(
    @inject(TagRepository) private readonly repository: TagRepository,
  ) {}

  public async execute(data?: ServiceDataType): Promise<FilterResultType<TagEntity>> {
    return await this.repository.find({
      page: data?.page,
      limit: data?.limit,
      q: data?.q,
    });
  }
}
