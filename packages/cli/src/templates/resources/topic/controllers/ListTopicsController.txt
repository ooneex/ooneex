import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import type { FilterResultType } from "@ooneex/types";
import { Assert } from "@ooneex/validation";
import { ListTopicsService } from "../services/ListTopicsService";
import type { TopicEntity } from "../entities/TopicEntity";

type ListTopicsRouteType = {
  query: { page?: number; limit?: number; q?: string };
  response: FilterResultType<TopicEntity>;
};

@Route.get("/topics", {
  name: "topic.list",
  version: 1,
  description: "List all topics",
  query: Assert({
    page: "number?",
    limit: "number?",
    q: "string?",
  }),
  roles: [ERole.USER],
})
export class ListTopicsController {
  constructor(
    @inject(ListTopicsService) private readonly service: ListTopicsService,
  ) {}

  public async index(context: ContextType<ListTopicsRouteType>) {
    const { page, limit, q } = context.query;

    const result = await this.service.execute({ page, limit, q });

    return context.response.json(result);
  }
}
