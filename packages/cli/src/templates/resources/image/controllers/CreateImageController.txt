import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import type { LocaleType } from "@ooneex/translation";
import { Assert } from "@ooneex/validation";
import type { ImageEntity } from "../entities/ImageEntity";
import { CreateImageService } from "../services/CreateImageService";

type CreateImageRouteType = {
  payload: {
    title: string;
    src: string;
    color?: string;
    description?: string;
    tags?: string[];
    topics?: string[];
    categories?: string[];
    status?: string;
    lang?: LocaleType;
  };
  response: ImageEntity | Record<string, never>;
};

@Route.post("/images", {
  name: "image.create",
  version: 1,
  description: "Create a new image",
  payload: Assert({
    title: "string",
    src: "string",
    color: "string?",
    description: "string?",
    tags: "string[]?",
    topics: "string[]?",
    categories: "string[]?",
    status: "string?",
    lang: "string?",
  }),
  response: Assert({
    id: "string",
    title: "string",
    src: "string",
    color: "string?",
    description: "string?",
    tags: "string[]?",
    topics: "string[]?",
    categories: "string[]?",
    status: "string?",
    lang: "string?",
  }),
  roles: [ERole.USER],
})
export class CreateImageController {
  constructor(
    @inject(CreateImageService) private readonly service: CreateImageService,
  ) {}

  public async index(context: ContextType<CreateImageRouteType>) {
    const { title, src, color, description, tags, topics, categories, status, lang } = context.payload;

    const image = await this.service.execute({ title, src, color, description, tags, topics, categories, status, lang });

    return context.response.json(image || {});
  }
}
