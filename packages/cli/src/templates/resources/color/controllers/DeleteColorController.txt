import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import { Assert } from "@ooneex/validation";
import { DeleteColorService } from "../services/DeleteColorService";

type DeleteColorRouteType = {
  params: { id: string };
  response: { id: string };
};

@Route.delete("/colors/:id", {
  name: "color.delete",
  version: 1,
  description: "Delete a color",
  params: {
    id: Assert("string"),
  },
  response: Assert({
    id: "string",
  }),
  roles: [ERole.USER],
})
export class DeleteColorController {
  constructor(
    @inject(DeleteColorService) private readonly service: DeleteColorService,
  ) {}

  public async index(context: ContextType<DeleteColorRouteType>) {
    const { id } = context.params;

    await this.service.execute({ id });

    return context.response.json({ id });
  }
}
