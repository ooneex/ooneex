import { inject } from "@ooneex/container";
import { decorator } from "@ooneex/service";
import type { IService } from "@ooneex/service";
import { TaskRepository } from "../repositories/TaskRepository";

type ServiceDataType = {
  id: string;
};

@decorator.service()
export class DeleteTaskService<T extends ServiceDataType = ServiceDataType> implements IService<T> {
  constructor(
    @inject(TaskRepository) private readonly repository: TaskRepository,
  ) {}

  public async execute(data?: T): Promise<void> {
    if (!data) return;

    await this.repository.delete({ id: data.id });
  }
}
