import { inject } from "@ooneex/container";
import { decorator } from "@ooneex/service";
import type { IService } from "@ooneex/service";
import { StatusEntity } from "../entities/StatusEntity";
import { StatusRepository } from "../repositories/StatusRepository";

type ServiceDataType = {
  id: string;
};

@decorator.service()
export class GetStatusService<T extends ServiceDataType = ServiceDataType> implements IService<T> {
  constructor(
    @inject(StatusRepository) private readonly repository: StatusRepository,
  ) {}

  public async execute(data?: T): Promise<StatusEntity | null> {
    if (!data) return null;

    return await this.repository.findOne(data.id);
  }
}
