#compdef oo ooneex

_oo_modules() {
  local -a modules
  if [[ -d modules ]]; then
    modules=(${(@f)"$(command ls -1 modules 2>/dev/null)"})
    compadd -a modules
  fi
}

_oo_route_names() {
  local -a names
  if [[ -d modules ]]; then
    names=(${(@f)"$(command grep -rh 'name:' modules/*/src/controllers/*Controller.ts 2>/dev/null | sed -n 's/.*name: "\([^"]*\.[^"]*\)".*/\1/p' | sort -u)"})
    compadd -a names
  fi
}

_oo_controllers() {
  local -a controllers
  if [[ -d modules ]]; then
    controllers=(${(@f)"$(command grep -rh 'export class' modules/*/src/controllers/*Controller.ts 2>/dev/null | sed -n 's/.*export class \([^ ]*Controller\).*/\1/p' | sort -u)"})
    compadd -a controllers
  fi
}

_oo_custom_commands() {
  local -a cmds
  if [[ -d modules ]]; then
    cmds=(${(@f)"$(command grep -rh -A1 'getName' modules/*/src/commands/*Command.ts 2>/dev/null | sed -n 's/.*return "\(.*\)".*/\1/p' | sort -u)"})
    compadd -a cmds
  fi
}

_oo() {
  local -a commands
  commands=(
    'app\:build:Build the application'
    'app\:start:Start the application'
    'app\:stop:Stop the application'
    'command\:run:Run a custom command from a module'
    'completion\:zsh:Install Zsh completion for oo command'
    'help:Show available commands'
    'make\:ai:Generate a new AI class'
    'make\:analytics:Generate a new analytics class'
    'make\:benchmark:Generate a new benchmark file'
    'make\:app:Generate a new application'
    'make\:cache:Generate a new cache class'
    'make\:claude\:skill:Generate Claude skills from templates'
    'make\:command:Generate a new command class'
    'make\:controller:Generate a new controller class'
    'make\:cron:Generate a new cron class'
    'make\:database:Generate a new database class'
    'make\:docker:Add a docker service to docker-compose.yml'
    'make\:entity:Generate a new entity class'
    'make\:logger:Generate a new logger class'
    'make\:mailer:Generate a new mailer class'
    'make\:middleware:Generate a new middleware class'
    'make\:migration:Generate a new migration file'
    'migration\:up:Run migrations for all modules'
    'make\:module:Generate a new module'
    'remove\:module:Remove an existing module'
    'make\:permission:Generate a new permission class'
    'make\:pubsub:Generate a new PubSub event class'
    'make\:release:Release packages with version bump, changelog, and git tag'
    'make\:repository:Generate a new repository class'
    'make\:resource\:book:Generate book resource (entity, migration, repository)'
    'make\:resource\:calendar-event:Generate calendar event resource (entity, migration, repository)'
    'make\:resource\:category:Generate category resource (entity, migration, repository)'
    'make\:resource\:color:Generate color resource (entity, migration, repository)'
    'make\:resource\:discount:Generate discount resource (entity, migration, repository)'
    'make\:resource\:folder:Generate folder resource (entity, migration, repository)'
    'make\:resource\:image:Generate image resource (entity, migration, repository)'
    'make\:resource\:note:Generate note resource (entity, migration, repository)'
    'make\:resource\:status:Generate status resource (entity, migration, repository)'
    'make\:resource\:tag:Generate tag resource (entity, migration, repository)'
    'make\:resource\:task:Generate task resource (entity, migration, repository)'
    'make\:resource\:topic:Generate topic resource (entity, migration, repository)'
    'make\:resource\:user:Generate user resource (entity, migration, repository)'
    'make\:resource\:video:Generate video resource (entity, migration, repository)'
    'make\:seed:Generate a new seed file'
    'seed\:run:Run seeds for all modules'
    'make\:service:Generate a new service class'
    'make\:storage:Generate a new storage class'
    'make\:vector-database:Generate a new vector database class'
  )

  _arguments -s \
    '1:command:->cmds' \
    '*::options:->opts' \
    && return 0

  case "$state" in
    cmds)
      _describe -t commands 'oo command' commands
      ;;
    opts)
      case "$words[1]" in
        command:run)
          _arguments -s \
            '1:command name:_oo_custom_commands'
          ;;
        make:benchmark)
          _arguments -s \
            '--name=[Benchmark name]:name:_oo_route_names' \
            '--module=[Module name]:module:_oo_modules' \
            '--target=[Target controller name]:target:_oo_controllers'
          ;;
        make:controller)
          _arguments -s \
            '--name=[Name of the resource]:name' \
            '--module=[Module name]:module:_oo_modules' \
            '--route-name=[Route name]:route_name' \
            '--route-path=[Route path]:route_path' \
            '--route-method=[Route HTTP method]:method:(GET POST PUT PATCH DELETE HEAD OPTIONS)' \
            '--is-socket[Socket mode]'
          ;;
        make:middleware)
          _arguments -s \
            '--name=[Name of the resource]:name' \
            '--module=[Module name]:module:_oo_modules' \
            '--is-socket[Socket mode]'
          ;;
        make:entity)
          _arguments -s \
            '--name=[Name of the resource]:name' \
            '--module=[Module name]:module:_oo_modules' \
            '--table-name=[Database table name]:table_name'
          ;;
        make:pubsub)
          _arguments -s \
            '--name=[Name of the resource]:name' \
            '--module=[Module name]:module:_oo_modules' \
            '--channel=[Channel name]:channel'
          ;;
        make:migration)
          _arguments -s \
            '--module=[Module name]:module:_oo_modules'
          ;;
        make:seed)
          _arguments -s \
            '--name=[Name of the resource]:name' \
            '--module=[Module name]:module:_oo_modules'
          ;;
        make:module)
          _arguments -s \
            '--name=[Name of the resource]:name'
          ;;
        remove:module)
          _arguments -s \
            '--name=[Module name]:name:_oo_modules'
          ;;
        make:docker)
          _arguments -s \
            '--name=[Docker service name]:name:(clickhouse elasticsearch grafana jaeger keycloak libretranslate maildev memcached minio mongodb mysql nats ooneex-jade postgres prometheus rabbitmq redis temporal vault)' \
            '--module=[Module name]:module:_oo_modules'
          ;;
        make:app)
          _arguments -s \
            '--name=[Name of the resource]:name' \
            '--destination=[Destination path]:destination:_files -/'
          ;;
        make:ai|make:analytics|make:cache|make:command|make:cron|make:database|make:logger|make:mailer|make:permission|make:repository|make:service|make:storage|make:vector:database)
          _arguments -s \
            '--name=[Name of the resource]:name' \
            '--module=[Module name]:module:_oo_modules'
          ;;
        migration:up|seed:run)
          _arguments -s \
            '--drop[Drop the database before running]'
          ;;
        app:build|app:start|app:stop|help|make:release|make:resource:book|make:resource:calendar-event|make:resource:category|make:resource:color|make:resource:discount|make:resource:folder|make:resource:image|make:resource:note|make:resource:status|make:resource:tag|make:resource:task|make:resource:topic|make:resource:user|make:resource:video|make:claude:skill|completion:zsh)
          ;;
      esac
      ;;
  esac
}

_oo "$@"
