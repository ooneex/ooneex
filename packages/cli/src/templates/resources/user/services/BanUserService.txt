import { ClerkAuth } from "@ooneex/auth";
import { inject } from "@ooneex/container";
import { decorator } from "@ooneex/service";
import type { IService } from "@ooneex/service";

type ServiceDataType = {
  userId: string;
  ban: boolean;
};

@decorator.service()
export class BanUserService<T extends ServiceDataType = ServiceDataType> implements IService<T> {
  constructor(@inject(ClerkAuth) private readonly auth: ClerkAuth) {}

  public async execute(data?: T): Promise<void> {
    if (!data) return;

    if (data.ban) {
      await this.auth.banUser(data.userId);
    } else {
      await this.auth.unbanUser(data.userId);
    }
  }
}
