import type { IVideo } from "@ooneex/video";
import type { LocaleType } from "@ooneex/translation";
import { random } from "@ooneex/utils";
import {
  Column,
  CreateDateColumn,
  Entity,
  PrimaryColumn,
  UpdateDateColumn,
} from "typeorm";

@Entity({
  name: "videos",
})
export class VideoEntity implements IVideo {
  @PrimaryColumn({ name: "id", type: "varchar", length: 25 })
  id: string = random.id();

  @Column({ name: "title", type: "varchar", length: 255 })
  title: string;

  @Column({ name: "src", type: "text" })
  src: string;

  @Column({ name: "thumbnail", type: "text" })
  thumbnail: string;

  @Column({ name: "color", type: "varchar", length: 50, nullable: true })
  color?: string | null;

  @Column({ name: "description", type: "text", nullable: true })
  description?: string | null;

  @Column({ name: "tags", type: "jsonb", nullable: true })
  tags?: string[] | null;

  @Column({ name: "topics", type: "jsonb", nullable: true })
  topics?: string[] | null;

  @Column({ name: "categories", type: "jsonb", nullable: true })
  categories?: string[] | null;

  @Column({ name: "status", type: "varchar", length: 50, nullable: true })
  status?: string | null;

  @Column({ name: "lang", type: "varchar", length: 10, nullable: true })
  lang?: LocaleType | null;

  @CreateDateColumn({ name: "created_at" })
  createdAt?: Date | null;

  @UpdateDateColumn({ name: "updated_at" })
  updatedAt?: Date | null;
}
