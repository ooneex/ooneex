import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import type { LocaleType } from "@ooneex/translation";
import { Assert } from "@ooneex/validation";
import type { FolderEntity } from "../entities/FolderEntity";
import { UpdateFolderService } from "../services/UpdateFolderService";

type UpdateFolderRouteType = {
  params: { id: string };
  payload: {
    name?: string;
    icon?: string;
    color?: string;
    tags?: string[];
    categories?: string[];
    lang?: LocaleType;
  };
  response: FolderEntity | Record<string, never>;
};

@Route.patch("/folders/:id", {
  name: "folder.update",
  version: 1,
  description: "Update a folder",
  params: {
    id: Assert("string"),
  },
  payload: Assert({
    name: "string?",
    icon: "string?",
    color: "string?",
    tags: "string[]?",
    categories: "string[]?",
    lang: "string?",
  }),
  response: Assert({
    id: "string",
    name: "string",
    icon: "string?",
    color: "string?",
    tags: "string[]?",
    categories: "string[]?",
    lang: "string?",
  }),
  roles: [ERole.USER],
})
export class UpdateFolderController {
  constructor(
    @inject(UpdateFolderService) private readonly service: UpdateFolderService,
  ) {}

  public async index(context: ContextType<UpdateFolderRouteType>) {
    const { id } = context.params;
    const { name, icon, color, tags, categories, lang } = context.payload;

    const folder = await this.service.execute({ id, name, icon, color, tags, categories, lang });

    return context.response.json(folder || {});
  }
}
