import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import { Assert } from "@ooneex/validation";
import { DeleteNoteService } from "../services/DeleteNoteService";

type DeleteNoteRouteType = {
  params: { id: string };
  response: { id: string };
};

@Route.delete("/notes/:id", {
  name: "note.delete",
  version: 1,
  description: "Delete a note",
  params: {
    id: Assert("string"),
  },
  response: Assert({
    id: "string",
  }),
  roles: [ERole.USER],
})
export class DeleteNoteController {
  constructor(
    @inject(DeleteNoteService) private readonly service: DeleteNoteService,
  ) {}

  public async index(context: ContextType<DeleteNoteRouteType>) {
    const { id } = context.params;

    await this.service.execute({ id });

    return context.response.json({ id });
  }
}
