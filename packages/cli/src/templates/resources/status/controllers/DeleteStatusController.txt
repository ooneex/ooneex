import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import { Assert } from "@ooneex/validation";
import { DeleteStatusService } from "../services/DeleteStatusService";

type DeleteStatusRouteType = {
  params: { id: string };
  response: { id: string };
};

@Route.Delete("/statuses/:id", {
  name: "status.delete",
  version: 1,
  description: "Delete a status",
  params: {
    id: Assert("string"),
  },
  response: Assert({
    id: "string",
  }),
  roles: [ERole.USER],
})
export class DeleteStatusController {
  constructor(
    @inject(DeleteStatusService) private readonly service: DeleteStatusService,
  ) {}

  public async index(context: ContextType<DeleteStatusRouteType>) {
    const { id } = context.params;

    await this.service.execute({ id });

    return context.response.json({ id });
  }
}
