import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import type { FilterResultType } from "@ooneex/types";
import { Assert } from "@ooneex/validation";
import { ListVideosService } from "../services/ListVideosService";
import type { VideoEntity } from "../entities/VideoEntity";

type ListVideosRouteType = {
  query: { page?: number; limit?: number; q?: string };
  response: FilterResultType<VideoEntity>;
};

@Route.get("/videos", {
  name: "video.list",
  version: 1,
  description: "List all videos",
  query: Assert({
    page: "number?",
    limit: "number?",
    q: "string?",
  }),
  roles: [ERole.USER],
})
export class ListVideosController {
  constructor(
    @inject(ListVideosService) private readonly service: ListVideosService,
  ) {}

  public async index(context: ContextType<ListVideosRouteType>) {
    const { page, limit, q } = context.query;

    const result = await this.service.execute({ page, limit, q });

    return context.response.json(result);
  }
}
