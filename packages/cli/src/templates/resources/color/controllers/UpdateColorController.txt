import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import type { LocaleType } from "@ooneex/translation";
import { Assert } from "@ooneex/validation";
import type { ColorEntity } from "../entities/ColorEntity";
import { UpdateColorService } from "../services/UpdateColorService";

type UpdateColorRouteType = {
  params: { id: string };
  payload: { name?: string; hex?: string; description?: string; lang?: LocaleType };
  response: ColorEntity | Record<string, never>;
};

@Route.patch("/colors/:id", {
  name: "color.update",
  version: 1,
  description: "Update a color",
  params: {
    id: Assert("string"),
  },
  payload: Assert({
    name: "string?",
    hex: "string?",
    description: "string?",
    lang: "string?",
  }),
  response: Assert({
    id: "string",
    name: "string",
    hex: "string",
    description: "string?",
    lang: "string?",
  }),
  roles: [ERole.USER],
})
export class UpdateColorController {
  constructor(
    @inject(UpdateColorService) private readonly service: UpdateColorService,
  ) {}

  public async index(context: ContextType<UpdateColorRouteType>) {
    const { id } = context.params;
    const { name, hex, description, lang } = context.payload;

    const color = await this.service.execute({ id, name, hex, description, lang });

    return context.response.json(color || {});
  }
}
