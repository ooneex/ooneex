import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import { Assert } from "@ooneex/validation";
import { DeleteUserProfileImageService } from "../services/DeleteUserProfileImageService";

type DeleteUserProfileImageRouteType = {
  params: { id: string };
  response: { id: string };
};

@Route.delete("/users/:id/profile/image", {
  name: "user.profile.image.delete",
  version: 1,
  description: "Delete a user profile image",
  params: {
    id: Assert("string"),
  },
  response: Assert({
    id: "string",
  }),
  roles: [ERole.USER],
})
export class DeleteUserProfileImageController {
  constructor(
    @inject(DeleteUserProfileImageService) private readonly service: DeleteUserProfileImageService,
  ) {}

  public async index(context: ContextType<DeleteUserProfileImageRouteType>) {
    const { id } = context.params;

    await this.service.execute({ userId: id });

    return context.response.json({ id });
  }
}
