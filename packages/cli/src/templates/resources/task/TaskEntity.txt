import type { LocaleType } from "@ooneex/translation";

import { random } from "@ooneex/utils";
import {
  Column,
  CreateDateColumn,
  Entity,
  PrimaryColumn,
  UpdateDateColumn,
} from "typeorm";

export type TaskStatusType = "Inbox" | "Todo" | "In Progress" | "Completed";

@Entity({
  name: "tasks",
})
export class TaskEntity {
  @PrimaryColumn({ name: "id", type: "varchar", length: 25 })
  id: string = random.id();

  @Column({ name: "title", type: "varchar", length: 255 })
  title: string;

  @Column({ name: "description", type: "text", nullable: true })
  description?: string | null;

  @Column({ name: "status", type: "varchar", length: 50, nullable: true })
  status?: TaskStatusType | null;

  @Column({ name: "priority", type: "varchar", length: 50, nullable: true })
  priority?: string | null;

  @Column({ name: "icon", type: "varchar", length: 255, nullable: true })
  icon?: string | null;

  @Column({ name: "color", type: "varchar", length: 50, nullable: true })
  color?: string | null;

  @Column({ name: "tags", type: "simple-array", nullable: true })
  tags?: string[] | null;

  @Column({ name: "categories", type: "simple-array", nullable: true })
  categories?: string[] | null;

  @Column({ name: "lang", type: "varchar", length: 10, nullable: true })
  lang?: LocaleType | null;

  @CreateDateColumn({ name: "created_at" })
  createdAt?: Date | null;

  @UpdateDateColumn({ name: "updated_at" })
  updatedAt?: Date | null;
}
