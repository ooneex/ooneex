import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import type { FilterResultType } from "@ooneex/types";
import { Assert } from "@ooneex/validation";
import { ListTagsService } from "../services/ListTagsService";
import type { TagEntity } from "../entities/TagEntity";

type ListTagsRouteType = {
  query: { page?: number; limit?: number; q?: string };
  response: FilterResultType<TagEntity>;
};

@Route.get("/tags", {
  name: "tag.list",
  version: 1,
  description: "List all tags",
  query: Assert({
    page: "number?",
    limit: "number?",
    q: "string?",
  }),
  roles: [ERole.USER],
})
export class ListTagsController {
  constructor(
    @inject(ListTagsService) private readonly service: ListTagsService,
  ) {}

  public async index(context: ContextType<ListTagsRouteType>) {
    const { page, limit, q } = context.query;

    const result = await this.service.execute({ page, limit, q });

    return context.response.json(result);
  }
}
