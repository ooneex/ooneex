import { inject } from "@ooneex/container";
import { decorator } from "@ooneex/service";
import type { IService } from "@ooneex/service";
import type { LocaleType } from "@ooneex/translation";
import { FolderEntity } from "../entities/FolderEntity";
import { FolderRepository } from "../repositories/FolderRepository";

type ServiceDataType = {
  id: string;
  name?: string;
  icon?: string;
  color?: string;
  tags?: string[];
  categories?: string[];
  lang?: LocaleType;
};

@decorator.service()
export class UpdateFolderService implements IService {
  constructor(
    @inject(FolderRepository) private readonly repository: FolderRepository,
  ) {}

  public async execute(data?: ServiceDataType): Promise<FolderEntity | null> {
    if (!data) return null;

    const { id, ...fields } = data;

    await this.repository.update({ id, ...fields });

    return await this.repository.findOne(id);
  }
}
