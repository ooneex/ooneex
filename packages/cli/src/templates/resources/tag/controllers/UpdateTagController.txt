import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import type { LocaleType } from "@ooneex/translation";
import { Assert } from "@ooneex/validation";
import { UpdateTagService } from "../services/UpdateTagService";

type UpdateTagRouteType = {
  params: { id: string };
  payload: { name?: string; lang?: LocaleType; color?: string; description?: string };
  response: { id: string; name: string; lang: LocaleType; color?: string; description?: string };
};

@Route.patch("/tags/:id", {
  name: "tag.update",
  version: 1,
  description: "Update a tag",
  params: {
    id: Assert("string"),
  },
  payload: Assert({
    name: "string?",
    lang: "string?",
    color: "string?",
    description: "string?",
  }),
  response: Assert({
    id: "string",
    name: "string",
    lang: "string",
    color: "string?",
    description: "string?",
  }),
  roles: [ERole.USER],
})
export class UpdateTagController {
  constructor(
    @inject(UpdateTagService) private readonly service: UpdateTagService,
  ) {}

  public async index(context: ContextType<UpdateTagRouteType>) {
    const { id } = context.params;
    const { name, lang, color, description } = context.payload;

    const tag = await this.service.execute({ id, name, lang, color, description });

    return context.response.json(tag);
  }
}
