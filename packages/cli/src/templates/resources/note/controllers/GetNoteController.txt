import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import { Assert } from "@ooneex/validation";
import type { NoteEntity } from "../entities/NoteEntity";
import { GetNoteService } from "../services/GetNoteService";

type GetNoteRouteType = {
  params: { id: string };
  response: NoteEntity | null;
};

@Route.get("/notes/:id", {
  name: "note.get",
  version: 1,
  description: "Get a note by id",
  params: {
    id: Assert("string"),
  },
  roles: [ERole.USER],
})
export class GetNoteController {
  constructor(
    @inject(GetNoteService) private readonly service: GetNoteService,
  ) {}

  public async index(context: ContextType<GetNoteRouteType>) {
    const { id } = context.params;

    const note = await this.service.execute({ id });

    return context.response.json(note);
  }
}
