import { ClerkAuth } from "@ooneex/auth";
import { inject } from "@ooneex/container";
import { decorator } from "@ooneex/service";
import type { IService } from "@ooneex/service";
import { UserRepository } from "../repositories/UserRepository";

type ServiceDataType = {
  userId: string;
  file: Blob | File;
};

@decorator.service()
export class UpdateUserProfileImageService<T extends ServiceDataType = ServiceDataType> implements IService<T> {
  constructor(
    @inject(ClerkAuth) private readonly auth: ClerkAuth,
    @inject(UserRepository) private readonly repository: UserRepository,
  ) {}

  public async execute(data?: T): Promise<void> {
    if (!data) return;

    const user = await this.auth.updateUserProfileImage(data.userId, {
      file: data.file,
    });
    await this.repository.updateProfile(data.userId, { avatar: user.imageUrl });
  }
}
