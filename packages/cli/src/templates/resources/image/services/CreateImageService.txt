import { inject } from "@ooneex/container";
import { decorator } from "@ooneex/service";
import type { IService } from "@ooneex/service";
import type { LocaleType } from "@ooneex/translation";
import { ImageEntity } from "../entities/ImageEntity";
import { ImageRepository } from "../repositories/ImageRepository";

type ServiceDataType = {
  title: string;
  src: string;
  color?: string;
  description?: string;
  tags?: string[];
  topics?: string[];
  categories?: string[];
  status?: string;
  lang?: LocaleType;
};

@decorator.service()
export class CreateImageService<T extends ServiceDataType = ServiceDataType> implements IService<T> {
  constructor(
    @inject(ImageRepository) private readonly repository: ImageRepository,
  ) {}

  public async execute(data?: T): Promise<ImageEntity | null> {
    if (!data) return null;

    const image = new ImageEntity();
    image.title = data.title;
    image.src = data.src;
    image.color = data.color;
    image.description = data.description;
    image.tags = data.tags;
    image.topics = data.topics;
    image.categories = data.categories;
    image.status = data.status;
    image.lang = data.lang;

    return await this.repository.create(image);
  }
}
