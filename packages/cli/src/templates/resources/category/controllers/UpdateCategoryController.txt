import type { ContextType } from "@ooneex/controller";
import { inject } from "@ooneex/container";
import { ERole } from "@ooneex/role";
import { Route } from "@ooneex/routing";
import type { LocaleType } from "@ooneex/translation";
import { Assert } from "@ooneex/validation";
import { UpdateCategoryService } from "../services/UpdateCategoryService";

type UpdateCategoryRouteType = {
  params: { id: string };
  payload: { name?: string; lang?: LocaleType; color?: string; description?: string };
  response: { id: string; name: string; lang: LocaleType; color?: string; description?: string };
};

@Route.patch("/categories/:id", {
  name: "category.update",
  version: 1,
  description: "Update a category",
  params: {
    id: Assert("string"),
  },
  payload: Assert({
    name: "string?",
    lang: "string?",
    color: "string?",
    description: "string?",
  }),
  response: Assert({
    id: "string",
    name: "string",
    lang: "string",
    color: "string?",
    description: "string?",
  }),
  roles: [ERole.USER],
})
export class UpdateCategoryController {
  constructor(
    @inject(UpdateCategoryService) private readonly service: UpdateCategoryService,
  ) {}

  public async index(context: ContextType<UpdateCategoryRouteType>) {
    const { id } = context.params;
    const { name, lang, color, description } = context.payload;

    const category = await this.service.execute({ id, name, lang, color, description });

    return context.response.json(category);
  }
}
